            )

            print(f"–°—Ç–∞—Ç—É—Å: {status['status']} - {status['progress']}%")

            time.sleep(poll_interval)
        
        # 3. –ü–æ–ª—É—á–µ–Ω–∏–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
        response = requests.get(
            f"{self.base_url}/api/v1/tasks/{task_id}/result",
            headers=self.headers
        )
        
        result = response.json()
        print(f"‚úì –ì–æ—Ç–æ–≤–æ! –í—Ä–µ–º—è –æ–±—Ä–∞–±–æ—Ç–∫–∏: {result['processing_time']:.1f}—Å")
        
        return result

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
